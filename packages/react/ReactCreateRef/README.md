import type {RefObject} from 'shared/ReactTypes';

// an immutable object with a single mutable value
// 只是返回了一个带 current 字段的对象，标签中的 ref 属性会挂载到 current 字段上
export function createRef(): RefObject {
  const refObject = {
    current: null,
  };
  if (__DEV__) {
    Object.seal(refObject);
  }
  return refObject;
}
